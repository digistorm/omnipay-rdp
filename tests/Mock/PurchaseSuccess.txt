function containsGETParameter(urlString) {
	var indexOfQuestionMark = urlString.indexOf('?');
	if(indexOfQuestionMark > -1){
		var indexOfEqualSign = urlString.indexOf('=');
		return (indexOfEqualSign > indexOfQuestionMark);
	}
	return false;
}
function convertToActionLink(formName) {
	var actionLink;
	if (typeof document.forms[formName] != "undefined") {
		actionLink = document.forms[formName].action;
		actionLink += containsGETParameter(document.forms[formName].action)?'&':'?';

		var inputs = document.forms[formName].getElementsByTagName('input');

		for (var i = inputs.length - 1; i >= 0; i--) {
			actionLink += inputs[i].name + "=" + encodeURI(inputs[i].value) + "&";
		}

		actionLink += actionLink.substring(0, actionLink.length-1);
	} 

	return actionLink;
}

function AutoSubmit() {
	var actionLink;

	if (typeof document.frmProcessPayment.action != 'undefined') {
		var formAction = document.frmProcessPayment.action;

		if (containsGETParameter(formAction)) {
			var parameters = formAction.substr(formAction.indexOf("?")+1).split('&')

			for (var i = 0; i < parameters.length; i++) {
				var paramObj = parameters[i].split('=');
				var key = paramObj[0];
				var val = paramObj[1];

				if (key == 'rdp_fm' && val == '2') {
					document.frmProcessPayment.method = 'get'
				} else if (key == 'rdp_fm' && val == '3') {
					actionLink = convertToActionLink('frmProcessPayment');
				}
			}

			if (typeof actionLink != 'undefined' && actionLink.length > 0) {
				window.location = actionLink;

				return false;
			}
		}
	}

	window.document.frmProcessPayment.submit();
	return true;
